<script>
  const gameEl = document.getElementById('game');
  const scoreEl = document.getElementById('score');
  const livesEl = document.getElementById('lives');
  const infoEl = document.getElementById('info');
  const menuEl = document.getElementById('menu');
  const gameOverEl = document.getElementById('gameOver');
  const messageEl = document.getElementById('message');
  const popSound = document.getElementById('popSound');

  let score = 0;
  let lives = 2;
  let mode = "facile";
  let speed = 5000;
  let maxLives = 2;
  let spawnInterval = null;
  let activeBlocks = new Map(); // Map pour garder les blocks actifs et leurs timers

  const mechanics = ['explosive', 'frozen', 'bomb'];

  function startGame(selectedMode) {
    mode = selectedMode;
    if (mode === "facile") { speed = 5000; maxLives = 2; }
    else if (mode === "moyen") { speed = 3000; maxLives = 1; }
    else if (mode === "impossible") { speed = 1500; maxLives = 1; }

    score = 0;
    lives = maxLives;

    scoreEl.textContent = score;
    livesEl.textContent = lives;
    gameEl.innerHTML = '';
    infoEl.style.display = 'block';
    menuEl.style.display = 'none';
    gameOverEl.style.display = 'none';
    messageEl.textContent = '';

    activeBlocks.forEach(({timeout}) => clearTimeout(timeout));
    activeBlocks.clear();

    spawnInterval = setInterval(spawnBlock, speed / 2);
  }

  function spawnBlock() {
    // Limiter le nombre de blocks max √† 6 (nombre de colonnes)
    if (activeBlocks.size >= 6) return;

    const block = document.createElement('div');
    block.classList.add('block');

    let isSpecial = Math.random() < 0.5;
    if (isSpecial) {
      let type = mechanics[Math.floor(Math.random() * mechanics.length)];
      block.classList.add(type);
      block.dataset.type = type;
    } else {
      block.dataset.type = 'normal';
    }

    gameEl.appendChild(block);

    // Timer de disparition du bloc
    const timeout = setTimeout(() => {
      if (gameEl.contains(block)) {
        loseLife();
        removeBlock(block);
      }
    }, speed);

    // Ajouter √† la map activeBlocks
    activeBlocks.set(block, {timeout});

    block.addEventListener('click', () => {
      if (!gameEl.contains(block)) return;

      clearTimeout(activeBlocks.get(block).timeout);
      activeBlocks.delete(block);

      if (block.dataset.type !== 'normal') {
        popSound.currentTime = 0;
        popSound.play();
        score++;
        scoreEl.textContent = score;
      }
      // Si c‚Äôest un mauvais bloc (normal), on ne fait rien ni en bien ni en mal

      removeBlock(block);
    });
  }

  function removeBlock(block) {
    if (gameEl.contains(block)) {
      gameEl.removeChild(block);
    }
  }

  function loseLife() {
    lives--;
    livesEl.textContent = lives;
    if (lives <= 0) {
      endGame();
    }
  }

  function endGame() {
    clearInterval(spawnInterval);
    activeBlocks.forEach(({timeout}, block) => clearTimeout(timeout));
    activeBlocks.clear();
    gameOverEl.style.display = 'block';
    messageEl.textContent = '';
  }

  function watchAd() {
    if (mode !== 'facile') {
      messageEl.textContent = "üö´ Pas de vie suppl√©mentaire en mode moyen ou impossible.";
      return;
    }
    messageEl.textContent = "üé• Tu as regard√© une pub (imagin√©e) ! Vie offerte.";
    lives = 1;
    livesEl.textContent = lives;
    gameOverEl.style.display = 'none';
    spawnInterval = setInterval(spawnBlock, speed / 2);
  }

  function restartGame() {
    menuEl.style.display = 'block';
    infoEl.style.display = 'none';
    gameEl.innerHTML = '';
    gameOverEl.style.display = 'none';
    messageEl.textContent = '';
  }
</script>
